\begin{frame}{条件付き確率}
    \begin{block}{条件付き確率の定義}
	全事象を$U$とする。$2$つの事象$A,B$について，条件付き確率$P_A(B)$は
	\begin{equation}
		P_A(B)=\frac{P(A\cap B)}{P(A)}
	\end{equation}
	と表される。（ただし，$P(A)\ne 0$）
    \end{block}
\end{frame}

\begin{frame}{乗法定理}
    \begin{block}{乗法定理}
	$2$つの事象$A,B$がともに起こる確率$P(A\cap B)$は
	\begin{equation}
		P(A\cap B)=P(A)\cdot P_A(B)
	\end{equation}
	と表される。（ただし，$P(A)\ne 0$）
    \end{block}
\end{frame}

	$2$つの事象$A,B$がともに起こる確率$P(A\cap B)$は
	\begin{equation}
		P(A\cap B)=P(A)\cdot P_A(B)
	\end{equation}
	と表される。（ただし，$P(A)\ne 0$）
    \end{block}

乗法定理より
\begin{equation}
P(A\cap B)=P(A)\cdot P_A(B)
\end{equation}

\end{frame}
